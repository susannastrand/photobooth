<script>
  // =============================================
  // PROJEKSJONSBESKRIVELSER
  // =============================================
  const projInfo = {
    none:      { name: "Original",            desc: "Slik ser virkeligheten ut – ingen forvrengning." },
    mercator:  { name: "Mercator (1569)",      desc: "Brukt i Google Maps. Bevarer vinkler, men blåser opp størrelsen nær polene voldsomt. Grønland ser like stort ut som Afrika – selv om Afrika er 14× større." },
    peters:    { name: "Peters / Equal-area (1973)", desc: "Bevarer arealer korrekt, men strekker ut form mot polene og klemmer inn mot ekvator. En reaksjon på Mercators eurosentriske bias." },
    goode:     { name: "Goode's Homolosine (1923)", desc: "Klipper opp verdenskartet langs havene for å bevare landform. Du ser sjelden at kloden faktisk kan deles opp slik." },
    robinson:  { name: "Robinson (1963)",      desc: "Kompromiss mellom area og form – brukt av National Geographic til 1988. Hverken korrekt størrelse eller korrekt form, men visuelt behagelig." },
    azimuthal: { name: "Azimutal (polprojeksjon)", desc: "Kartet sett fra et punkt rett ovenfra, som fra Nordpolen. Viser at all projeksjon er perspektivtaking – bokstavelig talt." }
  };


  // =============================================
  // OPPSETT
  // =============================================
  let currentProj = 'none';
  let snapped     = null;
  let showGrid    = true;


  const liveCanvas = document.getElementById('live');
  const outCanvas  = document.getElementById('output');
  const liveCtx    = liveCanvas.getContext('2d');
  const outCtx     = outCanvas.getContext('2d');
  const statusEl   = document.getElementById('status');


  const W = 340, H = 340;


  // =============================================
  // GRID-TOGGLE
  // =============================================
  document.getElementById('show-grid').addEventListener('change', e => {
    showGrid = e.target.checked;
    if (snapped) renderOutput('snap');
  });


  // =============================================
  // PROJEKSJONSLOGIKK (piksel-for-piksel)
  // Returnerer kilde-koordinater [sx, sy] i [-1, 1]
  // gitt destinasjons-koordinater [nx, ny] i [-1, 1]
  // =============================================
  function projectPixel(nx, ny, proj) {
    let sx = nx, sy = ny;


    if (proj === 'mercator') {
      const lat = ny * Math.PI * 0.45;
      sy = Math.log(Math.tan(Math.PI / 4 + lat / 2)) / 3.0;
      sx = nx;


    } else if (proj === 'peters') {
      sy = Math.sin(ny * Math.PI / 2) * 1.35;
      sx = nx / 1.1;


    } else if (proj === 'goode') {
      const cut = 0.3;
      if (Math.abs(nx) < cut) {
        sx = nx / cut * 0.6;
        sy = ny;
      } else {
        const sign = nx > 0 ? 1 : -1;
        const t = (Math.abs(nx) - cut) / (1 - cut);
        sx = sign * (0.6 + t * 0.4);
        sy = ny + sign * Math.sin(t * Math.PI) * 0.35;
      }


    } else if (proj === 'robinson') {
      const absY = Math.abs(ny);
      sx = nx * (1 - absY * absY * 0.3);
      sy = ny * (0.95 - absY * 0.12);


    } else if (proj === 'azimuthal') {
      const r = Math.sqrt(nx * nx + ny * ny);
      if (r > 1.0) return null;
      const theta = Math.atan2(ny, nx);
      const rNew  = Math.asin(r) / (Math.PI / 2);
      sx = Math.cos(theta) * rNew;
      sy = Math.sin(theta) * rNew;
    }


    return [sx, sy];
  }


  // =============================================
  // TEGN PROJISERT BILDE
  // =============================================
  function applyProjection(srcCtx, dstCtx, proj) {
    const src = srcCtx.getImageData(0, 0, W, H);
    const dst = dstCtx.createImageData(W, H);
    const sd = src.data, dd = dst.data;


    for (let py = 0; py < H; py++) {
      for (let px = 0; px < W; px++) {
        const nx = (px / W) * 2 - 1;
        const ny = (py / H) * 2 - 1;
        const result = projectPixel(nx, ny, proj);
        if (!result) { dd[(py * W + px) * 4 + 3] = 0; continue; }


        const [sx, sy] = result;
        const srcX = Math.round((sx + 1) / 2 * W);
        const srcY = Math.round((sy + 1) / 2 * H);


        if (srcX >= 0 && srcX < W && srcY >= 0 && srcY < H) {
          const si = (srcY * W + srcX) * 4;
          const di = (py * W + px) * 4;
          dd[di]     = sd[si];
          dd[di + 1] = sd[si + 1];
          dd[di + 2] = sd[si + 2];
          dd[di + 3] = sd[si + 3];
        }
      }
    }
    dstCtx.putImageData(dst, 0, 0);
  }


  // =============================================
  // TEGN KOORDINATGRID (graticule)
  // Linjene vrenger seg sammen med bildet
  // =============================================
  function drawGrid(ctx, proj) {
    // Hvilke breddegrader og lengdegrader som vises
    const LATLINES = [-60, -45, -30, -15, 0, 15, 30, 45, 60];
    const LONLINES = [-120, -90, -60, -30, 0, 30, 60, 90, 120];
    const STEPS    = 120; // antall punkter per linje (høyere = glattere kurver)


    // Hjelpe-funksjon: konverter normaliserte koordinater til lerret-piksler
    function toCanvas(nx, ny) {
      const result = proj === 'none' ? [nx, ny] : projectPixel(nx, ny, proj);
      if (!result) return null;
      return [(result[0] + 1) / 2 * W, (result[1] + 1) / 2 * H];
    }


    ctx.save();
    ctx.strokeStyle = 'rgba(255, 255, 255, 0.55)'; // farge på gridlinjene
    ctx.lineWidth   = 0.8;


    // Breddegrader (horisontale linjer)
    for (const lat of LATLINES) {
      const ny = lat / 90;
      ctx.beginPath();
      let started = false;
      for (let i = 0; i <= STEPS; i++) {
        const nx = -1 + (2 * i / STEPS);
        const pt = toCanvas(nx, ny);
        if (!pt) { started = false; continue; }
        if (!started) { ctx.moveTo(pt[0], pt[1]); started = true; }
        else ctx.lineTo(pt[0], pt[1]);
      }
      ctx.stroke();


      // Etikett
      const mid = toCanvas(0, ny);
      if (mid) {
        ctx.save();
        ctx.fillStyle = 'rgba(255, 255, 255, 0.85)';
        ctx.font      = '9px system-ui, sans-serif';
        ctx.textAlign = 'center';
        const label = lat === 0 ? 'Ekv.' : (lat > 0 ? lat + '°N' : Math.abs(lat) + '°S');
        ctx.fillText(label, mid[0], mid[1] - 3);
        ctx.restore();
      }
    }


    // Lengdegrader / meridianer (vertikale linjer)
    for (const lon of LONLINES) {
      const nx = lon / 180;
      ctx.beginPath();
      let started = false;
      for (let i = 0; i <= STEPS; i++) {
        const ny = -1 + (2 * i / STEPS);
        const pt = toCanvas(nx, ny);
        if (!pt) { started = false; continue; }
        if (!started) { ctx.moveTo(pt[0], pt[1]); started = true; }
        else ctx.lineTo(pt[0], pt[1]);
      }
      ctx.stroke();


      // Etikett
      const mid = toCanvas(nx, 0.05);
      if (mid) {
        ctx.save();
        ctx.fillStyle = 'rgba(255, 255, 255, 0.85)';
        ctx.font      = '9px system-ui, sans-serif';
        ctx.textAlign = 'center';
        const label = lon === 0 ? 'PM' : (lon > 0 ? lon + '°Ø' : Math.abs(lon) + '°V');
        ctx.fillText(label, mid[0] + 2, mid[1]);
        ctx.restore();
      }
    }


    ctx.restore();
  }


  // =============================================
  // RENDER OUTPUT-LERRET
  // =============================================
  function renderOutput(src) {
    const source = src === 'live' ? liveCanvas : snapped;
    if (currentProj === 'none') {
      outCtx.drawImage(source, 0, 0, W, H);
    } else {
      const tmp    = document.createElement('canvas');
      tmp.width    = W;
      tmp.height   = H;
      const tCtx   = tmp.getContext('2d');
      tCtx.drawImage(source, 0, 0, W, H);
      applyProjection(tCtx, outCtx, currentProj);
    }
    if (showGrid) drawGrid(outCtx, currentProj);
  }


  // =============================================
  // PROJEKSJONSKNAPPER – klikk
  // =============================================
  document.querySelectorAll('.proj-btn').forEach(btn => {
    btn.addEventListener('click', () => {
      document.querySelectorAll('.proj-btn').forEach(b => b.classList.remove('active'));
      btn.classList.add('active');
      currentProj = btn.dataset.proj;
      const info  = projInfo[currentProj];
      document.getElementById('proj-name').textContent = info.name;
      document.getElementById('proj-desc').textContent = info.desc;
      if (snapped) renderOutput('snap');
    });
  });


  // =============================================
  // KAMERA
  // =============================================
  const video = document.createElement('video');
  video.autoplay    = true;
  video.playsInline = true;
  video.muted       = true;


  navigator.mediaDevices.getUserMedia({ video: { facingMode: 'user', width: 340, height: 340 }, audio: false })
    .then(stream => {
      video.srcObject = stream;
      statusEl.textContent = 'Kamera aktivt – still deg i bildet';


      function drawLive() {
        liveCtx.save();
        liveCtx.scale(-1, 1);
        liveCtx.drawImage(video, -W, 0, W, H);
        liveCtx.restore();
        if (!snapped) renderOutput('live');
        requestAnimationFrame(drawLive);
      }
      video.addEventListener('loadedmetadata', drawLive);
    })
    .catch(() => {
      statusEl.textContent = 'Kameratilgang ikke tilgjengelig – last opp bilde';
      const snapBtn = document.getElementById('snap-btn');
      snapBtn.textContent = '📁 Last opp bilde';
      snapBtn.addEventListener('click', () => {
        const inp    = document.createElement('input');
        inp.type     = 'file';
        inp.accept   = 'image/*';
        inp.onchange = e => {
          const img  = new Image();
          img.onload = () => {
            liveCtx.drawImage(img, 0, 0, W, H);
            snapped        = document.createElement('canvas');
            snapped.width  = W;
            snapped.height = H;
            snapped.getContext('2d').drawImage(liveCanvas, 0, 0);
            renderOutput('snap');
            statusEl.textContent = 'Bilde lastet – velg en projeksjon';
          };
          img.src = URL.createObjectURL(e.target.files[0]);
        };
        inp.click();
      });
    });


  // =============================================
  // TA BILDE-KNAPP
  // =============================================
  document.getElementById('snap-btn').addEventListener('click', () => {
    if (!video.srcObject) return;
    snapped        = document.createElement('canvas');
    snapped.width  = W;
    snapped.height = H;
    const sc       = snapped.getContext('2d');
    sc.save();
    sc.scale(-1, 1);
    sc.drawImage(video, -W, 0, W, H);
    sc.restore();
    renderOutput('snap');
    statusEl.textContent = 'Bilde tatt! Prøv de ulike projeksjonene →';
  });
</script>
